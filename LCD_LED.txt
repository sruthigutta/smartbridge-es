#define F_CPU 16000000UL
#include <avr/io.h>
#include <util/delay.h>
#define RS 7
#define E 5
void send_command(unsigned char command);
void send_character(unsigned char character );
int main(void)
{
	DDRC=0XFF;
	DDRD=0XFF;
	DDRB=0XFF;
 while(1)
 {
	 PORTB=0XFF;
	send_command(0X01);
	send_command(0X38);
	send_command(0X0E);
	
	
	send_character(0X53);
	_delay_ms(1000);

	

PORTB=0X00;
	

	send_command(0X01);
	send_command(0X38);
	send_command(0X0E);
	
	
	send_character(0X6F);
	_delay_ms(1000);
}
}
void send_command(unsigned char command)
{
	PORTC=command;
	PORTD&=~(1<<RS);
	PORTD|=(1<<E);
	_delay_ms(50);
	 PORTD&=~(1<<E);
	 PORTC=0;
 }
 void send_character(unsigned char character)
 {
	 
	 PORTC=character;
	 PORTD|=(1<<RS);
	 PORTD|=(1<<E);
	 _delay_ms(50);
	 PORTD&=~(1<<E);
	 PORTC=0;
 }
