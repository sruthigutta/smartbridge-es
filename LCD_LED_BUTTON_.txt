 #ifndef F_CPU
 #define F_CPU 16000000UL
 #endif
 
 #include <avr/io.h>
 #include <util/delay.h>
 
 #define RS 7
 #define E 5
 
 void send_command(unsigned char command);
 void send_character(unsigned char character );
 
 int main(void)
 {
	 DDRC=0XFF;
	 DDRD=0XFF;
	 DDRB=0X00;
	 DDRA=0XFF;

	 send_command(0X01);
	 send_command(0X38);
	 send_command(0X0E);
	 _delay_ms(50);
	 
	while(1)
	{
		if((PINB &(1<<PB0))==0)
		{
			PORTA|=(1<<PA0);
			send_character(0X45);
	        _delay_ms(1000);
			send_command(0X01);
			
			PORTA&=~(1<<PA0);
			send_character(0X4C);
			_delay_ms(1000);
			send_command(0X01);
	 
		}
	}
 }
 
 void send_command(unsigned char command)
 {
	 PORTC=command;
	 PORTD&=~(1<<RS);
	 PORTD|=(1<<E);
	 _delay_ms(50);
	 PORTD&=~(1<<E);
	 PORTC=0;
 }
 void send_character(unsigned char character)
 {
	 
	 PORTC=character;
	 PORTD|=(1<<RS);
	 PORTD|=(1<<E);
	 _delay_ms(50);
	 PORTD&=~(1<<E);
	 PORTC=0;
 }
